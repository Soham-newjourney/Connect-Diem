-- 1ï¸âƒ£ Create Database
CREATE DATABASE IF NOT EXISTS connect_db;
USE connect_db;

-- 2ï¸âƒ£ Students Table
CREATE TABLE students (
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    roll_no VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    department VARCHAR(100),
    year INT,
    section VARCHAR(10),
    contact VARCHAR(15),
    profile_pic VARCHAR(255) DEFAULT 'default_student.png',
    join_date DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 3ï¸âƒ£ Faculty Table
CREATE TABLE faculty (
    faculty_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    department VARCHAR(100),
    designation VARCHAR(100),
    contact VARCHAR(15),
    profile_pic VARCHAR(255) DEFAULT 'default_faculty.png',
    join_date DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 4ï¸âƒ£ Admins Table
CREATE TABLE admins (
    admin_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    role ENUM('superadmin','moderator','staff') DEFAULT 'staff',
    profile_pic VARCHAR(255) DEFAULT 'default_admin.png',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 5ï¸âƒ£ Attendance Table
CREATE TABLE attendance (
    attendance_id INT AUTO_INCREMENT PRIMARY KEY,
    student_id INT NOT NULL,
    faculty_id INT NOT NULL,
    subject VARCHAR(100),
    attendance_date DATE,
    status ENUM('Present','Absent','Late') DEFAULT 'Present',
    remarks VARCHAR(255),
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (faculty_id) REFERENCES faculty(faculty_id) ON DELETE CASCADE
);

-- 6ï¸âƒ£ Marks Table
CREATE TABLE marks (
    mark_id INT AUTO_INCREMENT PRIMARY KEY,
    student_id INT NOT NULL,
    faculty_id INT,
    subject VARCHAR(100),
    exam_type ENUM('Midterm','Final','Assignment','Project'),
    score FLOAT,
    total FLOAT,
    remarks VARCHAR(255),
    recorded_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (faculty_id) REFERENCES faculty(faculty_id) ON DELETE SET NULL
);

-- 7ï¸âƒ£ Events Table
CREATE TABLE events (
    event_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    department VARCHAR(100),
    event_date DATE,
    event_time TIME,
    venue VARCHAR(150),
    organizer VARCHAR(100),
    banner_image VARCHAR(255),
    qr_code VARCHAR(255),
    feedback_form_link VARCHAR(255),
    created_by ENUM('admin','faculty') DEFAULT 'admin',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 8ï¸âƒ£ Event Registrations Table
CREATE TABLE event_registrations (
    registration_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    student_id INT NOT NULL,
    check_in_status ENUM('Pending','Checked-in') DEFAULT 'Pending',
    check_in_time DATETIME,
    feedback_submitted BOOLEAN DEFAULT 0,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE
);

-- 9ï¸âƒ£ Resources Table
CREATE TABLE resources (
    resource_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200),
    description TEXT,
    file_path VARCHAR(255),
    category VARCHAR(100),
    uploaded_by ENUM('faculty','admin') DEFAULT 'faculty',
    faculty_id INT,
    upload_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (faculty_id) REFERENCES faculty(faculty_id) ON DELETE SET NULL
);

-- ğŸ”Ÿ Doubts Table
CREATE TABLE doubts (
    doubt_id INT AUTO_INCREMENT PRIMARY KEY,
    student_id INT NOT NULL,
    faculty_id INT,
    subject VARCHAR(100),
    question TEXT NOT NULL,
    answer TEXT,
    status ENUM('Open','Answered','Closed') DEFAULT 'Open',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    answered_at DATETIME,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (faculty_id) REFERENCES faculty(faculty_id) ON DELETE SET NULL
);

-- 11ï¸âƒ£ Collaboration Table
CREATE TABLE collaboration (
    post_id INT AUTO_INCREMENT PRIMARY KEY,
    author_type ENUM('student','faculty') NOT NULL,
    author_id INT NOT NULL,
    title VARCHAR(200),
    content TEXT,
    attachment VARCHAR(255),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 12ï¸âƒ£ Notifications Table
CREATE TABLE notifications (
    notification_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(150),
    message TEXT,
    target ENUM('all','students','faculty','admins') DEFAULT 'all',
    created_by INT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 13ï¸âƒ£ Feedback Table
CREATE TABLE feedback (
    feedback_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT,
    student_id INT,
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comments TEXT,
    submitted_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE
);
